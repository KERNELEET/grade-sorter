include 'emu8086.inc'

JMP START

DATA SEGMENT
N DW ?
MARKS DB 1000 DUP (?) ;1000 IS THE MAXIMUM NUMBER OF STUDENTS
ID DB 1000 DUP (?)
MSG1 DB 'Enter the number of students (1-1000): ',0
MSG2 DB 0Dh,0Ah, 0Dh,0Ah, 'Enter the IDs of students (0-255): ',0
MSG3 DB 0Dh,0Ah, 0Dh,0Ah, 'Enter the marks of students (0-100): ',0
HR DB 0Dh,0Ah, 0Dh,0Ah, '******************* Sorted Marks ********************',0
MSG4 DB 0Dh,0Ah, 0Dh,0Ah, 'ID: ', 09H, 'MARKS:', 0
ERROR_MSG DB 0Dh,0Ah, 'Invalid input! Please enter valid numbers.', 0

DATA ENDS

CODE SEGMENT
ASSUME DS:DATA CS:CODE

START: MOV AX, DATA
MOV DS, AX ; Initialize data segment

; Defining functions for the library EMU8086
DEFINE_SCAN_NUM
DEFINE_PRINT_STRING
DEFINE_PRINT_NUM
DEFINE_PRINT_NUM_UNS

; Reading the number of students
LEA SI, MSG1
CALL PRINT_STRING
CALL SCAN_NUM ; The function puts the input in CX
MOV N, CX

; Validate number of students
CMP N, 1000
JG INVALID_INPUT

; Reading IDs of students
LEA SI, MSG2
CALL PRINT_STRING
MOV SI, 0

LOOP1: CALL SCAN_NUM
MOV ID[SI], CL
INC SI
PRINT 0AH ; Print newline
PRINT 0DH ; Return cursor to the beginning
CMP SI, N
JNE LOOP1

; Reading marks of students
LEA SI, MSG3
CALL PRINT_STRING
MOV SI, 0

LOOP2: CALL SCAN_NUM
MOV MARKS[SI], CL
INC SI
PRINT 0AH ; Print newline
PRINT 0DH ; Return cursor to the beginning
CMP SI, N
JNE LOOP2

; Sorting the students according to marks using bubble sort
DEC N ; Because we won't compare the last element
MOV CX, N ; CX as I

OUTER: MOV SI, 0 ; SI as J
INNER: MOV AL, MARKS[SI]
MOV DL, ID[SI]
INC SI
CMP MARKS[SI], AL
JB SKIP
XCHG AL, MARKS[SI]
MOV MARKS[SI-1], AL
XCHG DL, ID[SI]
MOV ID[SI-1], DL

SKIP: CMP SI, CX
JL INNER
LOOP OUTER

INC N ; Restore the value of N after sorting

; Display the sorted table of IDs and marks
LEA SI, HR
CALL PRINT_STRING
LEA SI, MSG4
CALL PRINT_STRING
PRINT 0AH ; Print newline
PRINT 0DH ; Return cursor to the beginning

MOV SI, 0

LOOP3: MOV AX, 0
MOV AL, ID[SI]
CALL PRINT_NUM_UNS
PRINT 09H ; Print tab
MOV AL, MARKS[SI]
CALL PRINT_NUM_UNS
PRINT 0AH ; Print newline
PRINT 0DH ; Return cursor to the beginning
INC SI
CMP SI, N
JNE LOOP3

JMP END_PROGRAM

INVALID_INPUT:
LEA SI, ERROR_MSG
CALL PRINT_STRING
JMP END_PROGRAM

END_PROGRAM:
RET

CODE ENDS
END START
